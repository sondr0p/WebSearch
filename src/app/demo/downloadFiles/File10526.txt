   <script type="text/javascript">
        $(function () {
            

            //++ If user types in left side, disable right side
            $("#LeftSide input, #LeftSide select").on("keyup change", function () {
                DisableRight();
            });

            //++ If user types in right side, disable left side
            $("#RightSide input, #RightSide select").on("keyup change", function () {
                DisableLeft();
            });
        });

        function DisableLeft() {
            var textFound = false;
            $("#RightSide input, #RightSide select").each(function () {
                if ($(this).val().length > 0) {
                    textFound = true;
                }
            });

            if (textFound) {
                //++ Mark required
                $("#RightSide input, #RightSide select").each(function () {
                    Required($(this).attr("id"));
                });
                //++ Disable left side
                $("#LeftSide input, #LeftSide select").attr("disabled", true);
            } else {
                //++ Mark not required
                $("#RightSide input, #RightSide select").each(function () {
                    NotRequired($(this).attr("id"));
                });
                //++ Enable left side
                $("#LeftSide input, #LeftSide select").removeAttr("disabled");
            }
        }

        function DisableRight() {
            var textFound = false;
            $("#LeftSide input, #LeftSide select").each(function () {
                if ($(this).val().length > 0) {
                    textFound = true;
                }
            });

            if (textFound) {
                //++ Mark required
                $("#LeftSide input, #LeftSide select").each(function () {
                    Required($(this).attr("id"));
                });
                //++ Disable right side
                $("#RightSide input, #RightSide select").attr("disabled", true);
            } else {
                //++ Mark not required
                $("#LeftSide input, #LeftSide select").each(function () {
                    NotRequired($(this).attr("id"));
                });
                //++ Enable right side
                $("#RightSide input, #RightSide select").removeAttr("disabled");
            }
        }

        function ShowInfo() {
            $('#LearnMore').modal("show");
        }

        function SendHome() {
            document.location = "http://www.msbn.ms.gov/";
        }

        function SetIndex(idx) {
            $.ajax({
                type: "POST",
                url: "/Verification/search.aspx/SetIndex",
                contentType: "application/json; charset=utf-8",
                dataType: "json",
                data: JSON.stringify({ "idx": idx }),
                success: function () {
                    document.location = "/Verification/result.aspx";
                }
            });
        }
    </script>
