        function displayTipForThis(tipCtlId, ctl, orientation) {
            var tipCtl = document.getElementById(tipCtlId);
            var ctlWidth = ctl.clientWidth + 8;

            if ((BrowserDetect.browser == "Explorer") && (BrowserDetect.version < 8)) {
                tipCtl.style.visibility = "visible";
                tipCtl.style.display = "block";
            }
            else {
                var offset = cumulativeOffset(ctl);

                if (orientation != 'right')
                    tipCtl.style.left = offset.left + ctlWidth + "px";
                else
                    tipCtl.style.left = (offset.left - 270) + "px";
                tipCtl.style.top = offset.top + "px";
                tipCtl.style.visibility = "visible";
                tipCtl.style.display = "block";
            }
        }
        var cumulativeOffset = function (element) {
            var top = 0, left = 0;
            do {
                top += element.offsetTop || 0;
                left += element.offsetLeft || 0;
                element = element.offsetParent;
            } while (element);

            return {
                top: top,
                left: left
            };
        };
